Problema de Lógica → Controle de Saldo para Saque e Depósito

(server.js)
app.post('/clientes/:id/depositar', (req,res) => {
    const { id } = req.params;
    const { valor } = req.body;
    console.log({id, valor});
    try
    {
        db.run(`UPDATE clientes SET saldo = saldo + ? WHERE id = ?`, [valor, id]);
        return res.status(200).json();
    }
    catch(err){
        console.log(err);
        return res.status(400).json(err);
    }
})

app.post('/clientes/:id/sacar', (req,res) => {
    const { id } = req.params;
    const { valor } = req.body;
    console.log({id, valor});
    try
    {
        db.run(`UPDATE clientes SET saldo = saldo - ? WHERE id = ?`, [valor, id]);
        return res.status(200).json();
    }
    catch(err){
        console.log(err);
        return res.status(400).json(err);
    }
})
Ausência de validação: permitia saques maiores que o saldo
Ausência de controle para valores nulos ou negativos nas operações de depósito e saque
Operação não atômica e vulnerável a concorrência
Falta de segurança na manipulação de dados financeiros
